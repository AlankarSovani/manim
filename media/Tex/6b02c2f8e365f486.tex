\begin{align*}
\begin{aligned}&\text{A sequence } \{x_n\} \text{ is Cauchy if } \\ &\forall \varepsilon > 0, \text{ there exists a } N \in \mathbb{N} \\ &\text{such that for all } m, n \in \mathbb{N}, m, n > N, \\ &|x_m - x_n| < \varepsilon\end{aligned}
\end{align*}
